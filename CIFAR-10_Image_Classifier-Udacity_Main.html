import torch.nn as nn
import torch.nn.functional as F

class Classifier(nn.Module):
    def __init__(self):
        super(Classifier, self).__init__()
        self.fc1 = nn.Linear(3072, 1500)
        self.fc2 = nn.Linear(1500, 800)
        self.fc3 = nn.Linear(800, 400)
        self.fc4 = nn.Linear(400, 200)
        self.fc5 = nn.Linear(200, 10)
        self.dropout = nn.Dropout(0.33)
        self.batchnorm1 = nn.BatchNorm1d(1500)  # Add BatchNorm1d for the fully connected layers
        self.batchnorm2 = nn.BatchNorm1d(800)
        self.batchnorm3 = nn.BatchNorm1d(400)
        self.batchnorm4 = nn.BatchNorm1d(200)

    def forward(self, inputs):
        inputs = F.relu(self.fc1(inputs))
        inputs = self.batchnorm1(inputs)  # Apply BatchNorm1d after the first linear layer
        inputs = self.dropout(inputs)
        
        inputs = F.relu(self.fc2(inputs))
        inputs = self.batchnorm2(inputs)  # Apply BatchNorm1d after the second linear layer
        inputs = self.dropout(inputs)
        
        inputs = F.relu(self.fc3(inputs))
        inputs = self.batchnorm3(inputs)  # Apply BatchNorm1d after the third linear layer
        inputs = self.dropout(inputs)
        
        inputs = F.relu(self.fc4(inputs))
        inputs = self.batchnorm4(inputs)  # Apply BatchNorm1d after the fourth linear layer
        inputs = self.dropout(inputs)
        
        inputs = F.log_softmax(self.fc5(inputs), dim=1)
        return inputs

        log_ps = model(images)
        loss = criterion(log_ps, labels)
        loss.backward()
        optimizer.step()

        running_loss += loss.item()
        
        # Store the average training loss for this epoch
        train_loss = running_loss / len(trainloader)
        train_losses.append(train_loss)

    print(f"Epoch {e+1} - Training loss: {running_loss/len(trainloader)}")
